# Function to remove highly correlated variables
def remove_highly_correlated(continuous_columns, threshold=0.9):
    corr_matrix = continuous_columns.corr().abs()
    upper_tri = corr_matrix.where(np.triu(np.ones(corr_matrix.shape), k=1).astype(np.bool))
    to_drop = [column for column in upper_tri.columns if any(upper_tri[column] > threshold)]
    continuous_columns = continuous_columns.drop(columns=to_drop)
    return continuous_columns

# Remove highly correlated variables
threshold = 0.9
